
```js
// JavaScript example
function greet(name) {
  return `Hello, ${name}!`
}

console.log(greet('World'))
```

```python
# Python example
def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(10))
```
